            // Quit code

        });

    public IEnumerator LoadGameScene() {
        yield return FaderManager.Instance.MakeFaderAppear();
        yield return FaderManager.Instance.MakeLoadingScreen();
        yield return SceneManager.LoadSceneAsync(1);
    }
